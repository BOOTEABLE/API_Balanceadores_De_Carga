events {}
http {
    upstream books_api { 
        # Ahora Nginx sabe exactamente a qué contenedor enviar la carga
        server books_service_instancia1:5000 weight=3;
        server books_service_instancia2:5000 weight=1;
        server books_service_instancia3:5000 weight=2;
    } 
    
    upstream users_api { server users_service:5000; }
    upstream orders_api { server orders_service:5000; }

    server {
        listen 80;

        location /api/books/ {
            proxy_pass http://books_api;
            add_header X-Backend-Server $upstream_addr;
        }

        location /api/users/ {
            proxy_pass http://users_api;
        }

        location /api/orders/ {
            proxy_pass http://orders_api;
        }
    }
}
